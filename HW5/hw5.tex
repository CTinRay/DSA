        int strLength = str1.length();
        Int upper_bound = strLength - 1;
        Int lower_bound = 0;
        while( upper_bound != lower_bound ){
          Int mid = ( upper_bound + lower_bound ) / 2;
          if( postfixHash( str1, strLength - mid ) == 
              postfixHash( str2, strLength - mid ) ){
            upper_bound = mid;
          }else{
            lower_bound = mid + 1;
          }
        }
